1. Creare repository și fișiere inițiale

2. Adăugare settings.xml și redenumire

echo "<settings></settings>" > settings.xml
git add settings.xml
git commit -m "Commit file settings.xml"
git mv settings.xml configuration.xml
git commit -m "Commit configuration.xml"
git commit --amend -m "Commit file configuration.xml"

3. Creare branch stage și hosts.xml

git checkout -b stage
git push origin stage
echo "<hosts></hosts>" > hosts.xml
git add hosts.xml
git commit -m "Commit file hosts.xml"
git push origin stage
git checkout main
git merge stage
git push origin main

4. Creare branch dev și env.txt

git checkout stage
git checkout -b dev
echo "ENV=development" > env.txt
git add env.txt
git commit -m "Commit file env.txt"
git checkout stage
git merge dev
git push origin stage

5. Redenumire branch dev în develop

git branch -m dev develop
git push origin -u develop
git push origin --delete dev
